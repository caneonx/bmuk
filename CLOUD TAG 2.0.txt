   2) BUILD TAG LIST DOM
   ========================= */
function buildTagList(filterText = "") {
  const taglist = document.getElementById("taglist");
  taglist.innerHTML = "";

  const f = filterText.trim().toLowerCase();
  const visible = TAGS.filter(t => !f || t.label.toLowerCase().includes(f));

  // Keep TagCanvas stable: if filter removes too many, keep at least a few
  const safe = visible.length >= 6 ? visible : TAGS.slice(0, 12);

  for (const t of safe) {
    const a = document.createElement("a");
    a.href = t.url || "#";
    a.target = "_blank";
    a.rel = "noopener noreferrer";
    a.textContent = t.label;

    // Weight by font size (TagCanvas reads it automatically)
    a.style.fontSize = `${t.w}px`;

    // Title shows on hover (TagCanvas tooltip if enabled)
    a.title = `${t.label} (weight ${t.w})`;

    taglist.appendChild(a);
  }

  document.getElementById("countPill").textContent =
    `${safe.length} tag${safe.length === 1 ? "" : "s"}`;

  return safe.length;
}

function startCloud() {
  resizeCloudCanvas();
  try { TagCanvas.Delete("cloudCanvas"); } catch(_) {}

  try {
    TagCanvas.Start("cloudCanvas", "taglist", state.opts);
    setStatus("running");
  } catch (e) {
    console.error("TagCanvas failed:", e);
    setStatus("error");
  }
}

function updateCloud() {
  startCloud();
}

  const pill = document.getElementById("statusPill");
  if (s === "paused") pill.textContent = "paused";
  else if (s === "error") pill.textContent = "error";
  else pill.textContent = "running";
}

function resizeCloudCanvas() {
  const canvas = document.getElementById("cloudCanvas");
  const rect = canvas.getBoundingClientRect();

  const dpr = Math.max(1, Math.min(2.5, window.devicePixelRatio || 1));
  const w = Math.max(320, Math.floor(rect.width));
  const h = Math.max(320, Math.floor(rect.height));

  canvas.width = Math.floor(w * dpr);
  canvas.height = Math.floor(h * dpr);
}

function wireUI() {
  const speed = document.getElementById("speed");
  const depth = document.getElementById("depth");
  const zoom  = document.getElementById("zoom");
  const outline = document.getElementById("outline");
  const shape = document.getElementById("shape");
  const search = document.getElementById("search");

  const speedVal = document.getElementById("speedVal");
  const depthVal = document.getElementById("depthVal");
  const zoomVal  = document.getElementById("zoomVal");
  const outVal   = document.getElementById("outVal");

  function applyLive() {
    speedVal.textContent = Number(speed.value).toFixed(2);
    depthVal.textContent = Number(depth.value).toFixed(2);
    zoomVal.textContent  = Number(zoom.value).toFixed(2);
    outVal.textContent   = Number(outline.value);

    state.opts.maxSpeed = Number(speed.value);
    state.opts.depth = Number(depth.value);
    state.opts.zoom = Number(zoom.value);
    state.opts.outlineThickness = Number(outline.value);

    try { TagCanvas.tc["cloudCanvas"]?.Zoom(state.opts.zoom); } catch(_) {}
    debounceRestart();
  }

  let restartTimer = null;
  function debounceRestart() {
    clearTimeout(restartTimer);
    restartTimer = setTimeout(() => {
      if (state.running) startCloud();
    }, 180);
  }

  speed.addEventListener("input", applyLive);
  depth.addEventListener("input", applyLive);
  zoom.addEventListener("input", applyLive);
  outline.addEventListener("input", applyLive);

  shape.addEventListener("change", () => {
    state.opts.shape = shape.value;
    updateCloud();
  });

  let searchTimer = null;
  search.addEventListener("input", () => {
    clearTimeout(searchTimer);
    searchTimer = setTimeout(() => {
      buildTagList(search.value);
      updateCloud();
    }, 120);
  });

  document.getElementById("toggle").addEventListener("click", () => {
    const btn = document.getElementById("toggle");
    const tc = TagCanvas.tc["cloudCanvas"];
    if (!tc) return;

    if (state.running) {
      try { TagCanvas.Pause("cloudCanvas"); } catch(_) {}
      state.running = false;
      btn.textContent = "Resume";
      setStatus("paused");
    } else {
      try { TagCanvas.Resume("cloudCanvas"); } catch(_) {}
      state.running = true;
      btn.textContent = "Pause";
      setStatus("running");
    }
  });

  document.getElementById("shuffle").addEventListener("click", () => {
    for (let i = TAGS.length - 1; i > 0; i--) {
      const j = (Math.random() * (i + 1)) | 0;
      [TAGS[i], TAGS[j]] = [TAGS[j], TAGS[i]];
    }
    buildTagList(search.value);
    updateCloud();
  });

  document.getElementById("reset").addEventListener("click", () => {
    search.value = "";
    shape.value = "sphere";
    speed.value = "0.06";
    depth.value = "0.90";
    zoom.value  = "1.00";
    outline.value = "2";

    state.opts.shape = "sphere";
    state.opts.maxSpeed = 0.06;
    state.opts.depth = 0.90;
    state.opts.zoom = 1.0;

    try { TagCanvas.Resume("cloudCanvas"); } catch(_) {}
  });

  document.getElementById("focusAI").addEventListener("click", () => {

  });

  applyLive();
}

/* =========================
   6) MENU TOGGLE (button + keyboard M)
   ========================= */
function wireMenuToggle() {
  const menuBtn = document.getElementById("menuToggleBtn");
  const panel = document.querySelector(".panel");
  if (!menuBtn || !panel) return;

  let menuVisible = true;

  function setMenuVisible(v) {
    menuVisible = !!v;
    panel.classList.toggle("hidden", !menuVisible);
    menuBtn.textContent = menuVisible ? "✖ Close" : "☰ Menu";
  }

  menuBtn.addEventListener("click", () => setMenuVisible(!menuVisible));

  // Keyboard shortcut: press M to toggle (ignore when typing in inputs)
  window.addEventListener("keydown", (e) => {
    const t = e.target;
    const typing = t && (t.tagName === "INPUT" || t.tagName === "TEXTAREA" || t.isContentEditable);
    if (typing) return;
    if (e.key === "m" || e.key === "M") setMenuVisible(!menuVisible);
  });

  // start visible
  setMenuVisible(true);
}

/* =========================
   7) STARFIELD BACKGROUND
   ========================= */
function startStars() {
  const canvas = document.getElementById("stars");
  const ctx = canvas.getContext("2d", { alpha: true });

  let w = 0, h = 0, dpr = 1;
  let stars = [];

  function resize() {
    dpr = Math.max(1, Math.min(2.5, window.devicePixelRatio || 1));
    w = canvas.width = Math.floor(innerWidth * dpr);
    h = canvas.height = Math.floor(innerHeight * dpr);
    canvas.style.width = innerWidth + "px";
    canvas.style.height = innerHeight + "px";

    const count = Math.floor((innerWidth * innerHeight) / 12000);
    stars = Array.from({ length: Math.max(80, Math.min(240, count)) }, () => ({
      x: Math.random() * w,
      y: Math.random() * h,
      r: (Math.random() * 1.6 + 0.2) * dpr,
      s: (Math.random() * 0.55 + 0.08) * dpr,
      a: Math.random() * 0.65 + 0.2
    }));
  }

  function tick() {
    ctx.clearRect(0,0,w,h);

    const grd = ctx.createRadialGradient(w*0.3, h*0.2, 0, w*0.3, h*0.2, Math.max(w,h)*0.8);
    grd.addColorStop(0, "rgba(0,255,255,0.06)");
    grd.addColorStop(0.35, "rgba(255,0,255,0.04)");
    grd.addColorStop(1, "rgba(0,0,0,0)");
    ctx.fillStyle = grd;
    ctx.fillRect(0,0,w,h);

    for (const s of stars) {
      s.y += s.s;
      if (s.y > h + 10) { s.y = -10; s.x = Math.random() * w; }
      ctx.globalAlpha = s.a;
      ctx.beginPath();
      ctx.arc(s.x, s.y, s.r, 0, Math.PI*2);
      ctx.fillStyle = "rgba(255,255,255,1)";
      ctx.fill();
    }

    ctx.globalAlpha = 1;
    requestAnimationFrame(tick);
  }

  resize();
  tick();
  window.addEventListener("resize", resize);
}
